    class Solution {
    public long solution(int price, int money, int count) {

        long answer  = 0; 

        for(int i=1; i<=count; i++) {            
            answer = answer + price * i; 
        }  

        answer = money-answer;

        if(answer<0) 
            answer= answer*-1;
        else if(answer>0) 
            answer=0;




        return answer;
    }
    }
